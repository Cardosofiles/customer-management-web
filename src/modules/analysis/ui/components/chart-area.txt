'use client'

import * as React from 'react'
import { Area, AreaChart, CartesianGrid, XAxis } from 'recharts'

import {
  Card,
  CardAction,
  CardContent,
  CardDescription,
  CardHeader,
  CardTitle,
} from '@/components/ui/card'
import {
  ChartContainer,
  ChartTooltip,
  ChartTooltipContent,
  type ChartConfig,
} from '@/components/ui/chart'
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/components/ui/select'
import { Skeleton } from '@/components/ui/skeleton'
import { ToggleGroup, ToggleGroupItem } from '@/components/ui/toggle-group'
import { useIsMobile } from '@/hooks/use-mobile'
import { useChartData } from '@/modules/analysis/hooks/use-chart-data'

const chartConfig = {
  visitors: {
    label: 'Clientes',
  },
  pessoaFisica: {
    label: 'Pessoa Física',
    color: 'var(--primary)',
  },
  pessoaJuridica: {
    label: 'Pessoa Jurídica',
    color: 'var(--chart-2)',
  },
} satisfies ChartConfig

export function ChartAreaInteractive() {
  const isMobile = useIsMobile()
  const [timeRange, setTimeRange] = React.useState(() => (isMobile ? '7d' : '90d'))
  const { data, isLoading, isError } = useChartData()

  const filteredData = React.useMemo(() => {
    if (!data) return []
    const referenceDate = new Date()
    let daysToSubtract = 90
    if (timeRange === '30d') daysToSubtract = 30
    else if (timeRange === '7d') daysToSubtract = 7
    const startDate = new Date(referenceDate)
    startDate.setDate(startDate.getDate() - daysToSubtract)
    return data.filter((item) => new Date(item.date) >= startDate)
  }, [data, timeRange])

  return (
    <Card className="@container/card">
      <CardHeader>
        <CardTitle>Panorâma de Cadastros</CardTitle>
        <CardDescription>
          <span className="hidden @[540px]/card:block">Filtre a informação desejada</span>
          <span className="@[540px]/card:hidden">Três meses atrás</span>
        </CardDescription>
        <CardAction>
          <ToggleGroup
            type="single"
            value={timeRange}
            onValueChange={setTimeRange}
            variant="outline"
            className="hidden *:data-[slot=toggle-group-item]:px-4! @[767px]/card:flex"
          >
            <ToggleGroupItem value="90d">Três meses atrás</ToggleGroupItem>
            <ToggleGroupItem value="30d">30 dias atrás</ToggleGroupItem>
            <ToggleGroupItem value="7d">7 dias atrás</ToggleGroupItem>
          </ToggleGroup>
          <Select value={timeRange} onValueChange={setTimeRange}>
            <SelectTrigger
              className="flex w-40 **:data-[slot=select-value]:block **:data-[slot=select-value]:truncate @[767px]/card:hidden"
              size="sm"
              aria-label="Select a value"
            >
              <SelectValue placeholder="Três meses atrás" />
            </SelectTrigger>
            <SelectContent className="rounded-xl">
              <SelectItem value="90d" className="rounded-lg">
                Três meses atrás
              </SelectItem>
              <SelectItem value="30d" className="rounded-lg">
                30 dias atrás
              </SelectItem>
              <SelectItem value="7d" className="rounded-lg">
                7 dias atrás
              </SelectItem>
            </SelectContent>
          </Select>
        </CardAction>
      </CardHeader>
      <CardContent className="px-2 pt-4 sm:px-6 sm:pt-6">
        {isLoading ? (
          <Skeleton className="aspect-auto h-[250px] w-full" />
        ) : isError ? (
          <div className="flex aspect-auto h-[250px] w-full items-center justify-center text-sm text-muted-foreground">
            Erro ao carregar dados do gráfico.
          </div>
        ) : filteredData.length === 0 ? (
          <div className="flex aspect-auto h-[250px] w-full items-center justify-center text-sm text-muted-foreground">
            Nenhum cadastro no período selecionado.
          </div>
        ) : (
          <ChartContainer config={chartConfig} className="aspect-auto h-[250px] w-full">
            <AreaChart data={filteredData}>
              <defs>
                <linearGradient id="fillPessoaFisica" x1="0" y1="0" x2="0" y2="1">
                  <stop offset="5%" stopColor="var(--color-pessoaFisica)" stopOpacity={1.0} />
                  <stop offset="95%" stopColor="var(--color-pessoaFisica)" stopOpacity={0.1} />
                </linearGradient>
                <linearGradient id="fillPessoaJuridica" x1="0" y1="0" x2="0" y2="1">
                  <stop offset="5%" stopColor="var(--color-pessoaJuridica)" stopOpacity={0.8} />
                  <stop offset="95%" stopColor="var(--color-pessoaJuridica)" stopOpacity={0.1} />
                </linearGradient>
              </defs>
              <CartesianGrid vertical={false} />
              <XAxis
                dataKey="date"
                tickLine={false}
                axisLine={false}
                tickMargin={8}
                minTickGap={32}
                tickFormatter={(value) =>
                  new Date(value).toLocaleDateString('pt-BR', {
                    month: 'short',
                    day: 'numeric',
                  })
                }
              />
              <ChartTooltip
                cursor={false}
                content={
                  <ChartTooltipContent
                    labelFormatter={(value) =>
                      new Date(value).toLocaleDateString('pt-BR', {
                        month: 'short',
                        day: 'numeric',
                      })
                    }
                    indicator="dot"
                  />
                }
              />
              <Area
                dataKey="pessoaJuridica"
                type="natural"
                fill="url(#fillPessoaJuridica)"
                stroke="var(--color-pessoaJuridica)"
                stackId="a"
              />
              <Area
                dataKey="pessoaFisica"
                type="natural"
                fill="url(#fillPessoaFisica)"
                stroke="var(--color-pessoaFisica)"
                stackId="a"
              />
            </AreaChart>
          </ChartContainer>
        )}
      </CardContent>
    </Card>
  )
}
